## 🐍 Python - *Data Cleaning & Preparation*

```python
"""
MARKETING ROI ANALYSIS - PYTHON ETL
Business Goal: Clean 50,000+ campaign records for accurate ROI calculation
Tools in Service of Business: Python helps us trust our data
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime

# ============================================
# 1. LOAD DATA - 50,000+ campaigns
# ============================================
print("🚀 Loading marketing campaign data...")
df = pd.read_csv('marketing_campaigns.csv')
print(f"📊 Dataset Shape: {df.shape}")  # (52,347, 15)
print(f"📅 Date Range: {df['campaign_date'].min()} to {df['campaign_date'].max()}")

# ============================================
# 2. DATA CLEANING - Because dirty data = wrong decisions
# ============================================
print("\n🧹 Cleaning data for accurate business insights...")

# Remove duplicates (duplicate campaigns = double counting spend)
initial_count = len(df)
df.drop_duplicates(subset=['campaign_id', 'date'], inplace=True)
dupes_removed = initial_count - len(df)
print(f"✅ Removed {dupes_removed} duplicate campaigns")

# Handle missing values
print(f"❓ Missing values before cleaning:")
print(df.isnull().sum())

# For numeric columns: fill with median (outliers don't skew median)
numeric_cols = ['spend', 'impressions', 'clicks', 'conversions']
for col in numeric_cols:
    df[col].fillna(df[col].median(), inplace=True)

# For categorical: fill with 'Unknown'
categorical_cols = ['channel', 'campaign_type', 'target_audience']
for col in categorical_cols:
    df[col].fillna('Unknown', inplace=True)

# ============================================
# 3. FEATURE ENGINEERING - Creating business metrics
# ============================================
print("\n⚙️ Creating business metrics from raw data...")

# Core marketing metrics (these answer business questions)
df['ctr'] = (df['clicks'] / df['impressions'] * 100).round(2)
df['conversion_rate'] = (df['conversions'] / df['clicks'] * 100).round(2)
df['cpc'] = (df['spend'] / df['clicks']).round(2)
df['cpa'] = (df['spend'] / df['conversions']).round(2)

# Revenue calculation (assuming average order value)
AVG_ORDER_VALUE = 150  # $ per conversion
df['revenue'] = df['conversions'] * AVG_ORDER_VALUE
df['roi'] = ((df['revenue'] - df['spend']) / df['spend'] * 100).round(2)

# ============================================
# 4. BUSINESS INSIGHTS - Quick Python analysis
# ============================================
print("\n💡 Quick Business Insights from Python:")

# Channel performance summary
channel_summary = df.groupby('channel').agg({
    'spend': 'sum',
    'revenue': 'sum',
    'conversions': 'sum',
    'campaign_id': 'count'
}).round(2)

channel_summary['roi'] = ((channel_summary['revenue'] - channel_summary['spend']) / 
                          channel_summary['spend'] * 100).round(2)

print("\n📊 Channel Performance Summary:")
print(channel_summary.sort_values('roi', ascending=False))

# Top performing campaigns
top_campaigns = df.nlargest(10, 'roi')[['campaign_name', 'channel', 'spend', 'revenue', 'roi']]
print("\n🏆 Top 10 Campaigns by ROI:")
print(top_campaigns)

# ============================================
# 5. VISUALIZE - Quick insights before Power BI
# ============================================
fig, axes = plt.subplots(2, 2, figsize=(15, 10))

# ROI by Channel
channel_roi = df.groupby('channel')['roi'].mean().sort_values()
axes[0,0].barh(channel_roi.index, channel_roi.values, color='#4285F4')
axes[0,0].set_title('Average ROI by Channel', fontsize=14)
axes[0,0].set_xlabel('ROI (%)')

# Spend Distribution
spend_by_channel = df.groupby('channel')['spend'].sum()
axes[0,1].pie(spend_by_channel.values, labels=spend_by_channel.index, autopct='%1.1f%%')
axes[0,1].set_title('Budget Allocation by Channel', fontsize=14)

# Conversion Rate Distribution
axes[1,0].hist(df['conversion_rate'], bins=30, color='#34A853', edgecolor='black')
axes[1,0].set_title('Conversion Rate Distribution', fontsize=14)
axes[1,0].set_xlabel('Conversion Rate (%)')

# ROI vs Spend Scatter
axes[1,1].scatter(df['spend'], df['roi'], alpha=0.5, color='#EA4335')
axes[1,1].set_title('ROI vs Spend', fontsize=14)
axes[1,1].set_xlabel('Spend ($)')
axes[1,1].set_ylabel('ROI (%)')
axes[1,1].axhline(y=0, color='black', linestyle='--', alpha=0.3)

plt.tight_layout()
plt.savefig('python_quick_insights.png', dpi=100)
print("\n✅ Quick insights visualization saved!")

# ============================================
# 6. EXPORT CLEAN DATA - Ready for SQL, AWS, Power BI
# ============================================
df.to_csv('marketing_data_clean.csv', index=False)
print("\n✅ Clean data exported for SQL/AWS/Power BI")

# Business Summary
print("\n" + "="*50)
print("🎯 BUSINESS SUMMARY (from Python)")
print("="*50)
print(f"Total Campaigns Analyzed: {len(df):,}")
print(f"Total Marketing Spend: ${df['spend'].sum():,.2f}")
print(f"Total Revenue Generated: ${df['revenue'].sum():,.2f}")
print(f"Overall ROI: {((df['revenue'].sum() - df['spend'].sum()) / df['spend'].sum() * 100):.1f}%")
print(f"Best Channel: {channel_summary['roi'].idxmax()} ({channel_summary['roi'].max():.1f}% ROI)")
print("="*50)
```

## 🗄️ SQL Server - *Structured Analysis*

```sql
/*
MARKETING ROI ANALYSIS - SQL SERVER QUERIES
Business Goal: Answer specific business questions with structured data
Focus: Channel performance, campaign effectiveness, budget optimization
*/

-- ============================================
-- 1. DATABASE SETUP - Marketing Data Warehouse
-- ============================================
CREATE DATABASE MarketingAnalytics;
GO

USE MarketingAnalytics;
GO

-- Main campaigns table
CREATE TABLE MarketingCampaigns (
    CampaignID INT PRIMARY KEY,
    CampaignName NVARCHAR(100),
    Channel VARCHAR(50),
    CampaignType VARCHAR(50),
    TargetAudience VARCHAR(50),
    StartDate DATE,
    EndDate DATE,
    Spend DECIMAL(12,2),
    Impressions INT,
    Clicks INT,
    Conversions INT,
    Revenue DECIMAL(12,2),
    ROI AS ((Revenue - Spend) / Spend * 100),  -- Computed column
    CTR AS (Clicks * 1.0 / Impressions * 100), -- Computed column
    CPA AS (Spend / NULLIF(Conversions, 0)),    -- Cost per acquisition
    CreatedDate DATETIME DEFAULT GETDATE()
);

-- Load data (from Python cleaned file)
BULK INSERT MarketingCampaigns
FROM 'C:\data\marketing_data_clean.csv'
WITH (
    FIRSTROW = 2,
    FIELDTERMINATOR = ',',
    ROWTERMINATOR = '\n',
    TABLOCK
);

-- ============================================
-- 2. BUSINESS QUESTION: Which channels give best ROI?
-- ============================================
SELECT 
    Channel,
    COUNT(*) AS CampaignCount,
    SUM(Spend) AS TotalSpend,
    SUM(Revenue) AS TotalRevenue,
    ROUND(AVG(ROI), 2) AS AvgROI,
    ROUND(AVG(CTR), 2) AS AvgCTR,
    ROUND(AVG(CONVERT(FLOAT, CPA)), 2) AS AvgCPA,
    -- Performance rating
    CASE 
        WHEN AVG(ROI) > 200 THEN '⭐ CHAMPION'
        WHEN AVG(ROI) BETWEEN 150 AND 200 THEN '✅ GOOD'
        WHEN AVG(ROI) BETWEEN 100 AND 150 THEN '🟡 NEEDS OPTIMIZATION'
        ELSE '🔴 UNDERPERFORMING'
    END AS PerformanceRating
FROM MarketingCampaigns
WHERE Spend > 0
GROUP BY Channel
ORDER BY AvgROI DESC;

-- ============================================
-- 3. BUSINESS QUESTION: Which campaigns waste money?
-- ============================================
WITH CampaignPerformance AS (
    SELECT 
        CampaignID,
        CampaignName,
        Channel,
        Spend,
        Revenue,
        ROI,
        -- Rank campaigns within each channel
        RANK() OVER (PARTITION BY Channel ORDER BY ROI DESC) AS RankInChannel
    FROM MarketingCampaigns
    WHERE Spend > 0
)
SELECT 
    CampaignName,
    Channel,
    Spend,
    Revenue,
    ROI,
    'Negative ROI - Review immediately' AS Recommendation
FROM CampaignPerformance
WHERE ROI < 0
ORDER BY ROI;

-- ============================================
-- 4. BUSINESS QUESTION: What's the optimal budget range?
-- ============================================
WITH BudgetBands AS (
    SELECT 
        CASE 
            WHEN Spend < 1000 THEN '1. <$1K'
            WHEN Spend BETWEEN 1000 AND 5000 THEN '2. $1K-$5K'
            WHEN Spend BETWEEN 5001 AND 10000 THEN '3. $5K-$10K'
            WHEN Spend BETWEEN 10001 AND 25000 THEN '4. $10K-$25K'
            ELSE '5. >$25K'
        END AS BudgetBand,
        Spend,
        ROI,
        Revenue
    FROM MarketingCampaigns
)
SELECT 
    BudgetBand,
    COUNT(*) AS CampaignCount,
    SUM(Spend) AS TotalSpend,
    AVG(ROI) AS AvgROI,
    -- Success rate (campaigns with ROI > 100%)
    SUM(CASE WHEN ROI > 100 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) AS SuccessRate
FROM BudgetBands
GROUP BY BudgetBand
ORDER BY BudgetBand;

-- ============================================
-- 5. BUSINESS QUESTION: Monthly trend analysis
-- ============================================
SELECT 
    YEAR(StartDate) AS Year,
    MONTH(StartDate) AS Month,
    COUNT(*) AS CampaignsRun,
    SUM(Spend) AS MonthlySpend,
    SUM(Revenue) AS MonthlyRevenue,
    AVG(ROI) AS AvgROI,
    -- Month-over-month change
    LAG(SUM(Revenue)) OVER (ORDER BY YEAR(StartDate), MONTH(StartDate)) AS PrevMonthRevenue,
    (SUM(Revenue) - LAG(SUM(Revenue)) OVER (ORDER BY YEAR(StartDate), MONTH(StartDate))) / 
        NULLIF(LAG(SUM(Revenue)) OVER (ORDER BY YEAR(StartDate), MONTH(StartDate)), 0) * 100 AS MoMGrowth
FROM MarketingCampaigns
GROUP BY YEAR(StartDate), MONTH(StartDate)
ORDER BY Year, Month;

-- ============================================
-- 6. STORED PROCEDURE: Campaign Performance Report
-- ============================================
CREATE PROCEDURE sp_GetCampaignPerformance
    @StartDate DATE,
    @EndDate DATE,
    @Channel VARCHAR(50) = NULL
AS
BEGIN
    SELECT 
        CampaignName,
        Channel,
        CampaignType,
        Spend,
        Revenue,
        ROUND(ROI, 2) AS ROI,
        ROUND(CTR, 2) AS CTR,
        ROUND(CONVERSIONS * 1.0 / NULLIF(CLICKS, 0) * 100, 2) AS ConversionRate,
        -- Efficiency score (ROI per dollar spent)
        ROUND(ROI / NULLIF(Spend, 0) * 1000, 4) AS EfficiencyScore
    FROM MarketingCampaigns
    WHERE StartDate BETWEEN @StartDate AND @EndDate
        AND (@Channel IS NULL OR Channel = @Channel)
    ORDER BY ROI DESC;
END;
GO

-- Execute: EXEC sp_GetCampaignPerformance '2026-01-01', '2026-03-01', 'Google Search';

-- ============================================
-- 7. BUSINESS INSIGHTS VIEW - For Power BI
-- ============================================
CREATE VIEW vw_MarketingDashboard AS
SELECT 
    CampaignID,
    CampaignName,
    Channel,
    CampaignType,
    TargetAudience,
    StartDate,
    Spend,
    Revenue,
    ROI,
    CTR,
    CPA,
    -- Season (for trend analysis)
    CASE 
        WHEN MONTH(StartDate) IN (12,1,2) THEN 'Winter'
        WHEN MONTH(StartDate) IN (3,4,5) THEN 'Spring'
        WHEN MONTH(StartDate) IN (6,7,8) THEN 'Summer'
        ELSE 'Fall'
    END AS Season,
    -- ROI Category
    CASE 
        WHEN ROI > 200 THEN 'High Performer'
        WHEN ROI BETWEEN 100 AND 200 THEN 'Good'
        WHEN ROI BETWEEN 0 AND 100 THEN 'Average'
        ELSE 'Poor'
    END AS ROICategory
FROM MarketingCampaigns;
GO

-- ============================================
-- 8. BUSINESS SUMMARY REPORT
-- ============================================
SELECT 
    '💰 Total Marketing Spend' AS Metric, 
    CONCAT('$', FORMAT(SUM(Spend), 'N0')) AS Value
FROM MarketingCampaigns
UNION ALL
SELECT '📈 Total Revenue Generated', CONCAT('$', FORMAT(SUM(Revenue), 'N0'))
FROM MarketingCampaigns
UNION ALL
SELECT '📊 Average ROI', CONCAT(FORMAT(AVG(ROI), 'N1'), '%')
FROM MarketingCampaigns
UNION ALL
SELECT '🎯 Best Performing Channel', 
    (SELECT TOP 1 Channel FROM MarketingCampaigns GROUP BY Channel ORDER BY AVG(ROI) DESC)
UNION ALL
SELECT '⚠️ Campaigns with Negative ROI', 
    CONCAT(COUNT(*), ' campaigns')
FROM MarketingCampaigns
WHERE ROI < 0;
```

```python
"""
MARKETING ROI ANALYSIS - AWS IMPLEMENTATION
Business Goal: Enable team-wide access to marketing data, run ad-hoc queries
Focus: Data lake on S3, serverless queries with Athena
"""

import boto3
import pandas as pd
import json
from io import StringIO
import time

# ============================================
# 1. AWS S3 SETUP - Central Data Lake
# ============================================
print("🚀 Setting up AWS S3 data lake...")

# Configure AWS (using credentials from environment)
s3_client = boto3.client(
    's3',
    region_name='us-east-1'  # Virginia - good for cost
)

# Create bucket for marketing data
bucket_name = 'marketing-roi-analytics-6663'
try:
    s3_client.create_bucket(
        Bucket=bucket_name
    )
    print(f"✅ Created bucket: {bucket_name}")
except:
    print(f"ℹ️ Bucket {bucket_name} already exists")

# ============================================
# 2. ORGANIZE DATA IN S3 (Like folders for business units)
# ============================================
print("\n📁 Organizing marketing data in S3...")

# Structure: s3://bucket/category/year/month/file.csv
folders = [
    'raw/2026/01/',
    'raw/2026/02/',
    'raw/2026/03/',
    'processed/daily/',
    'processed/monthly/',
    'reports/dashboard/',
    'reports/executive/'
]

for folder in folders:
    s3_client.put_object(
        Bucket=bucket_name,
        Key=folder
    )
print("✅ Folder structure created")

# ============================================
# 3. UPLOAD DATA TO S3
# ============================================
print("\n📤 Uploading marketing data to S3...")

# Read cleaned data from Python
df = pd.read_csv('marketing_data_clean.csv')

# Convert to CSV in memory
csv_buffer = StringIO()
df.to_csv(csv_buffer, index=False)

# Upload raw data
s3_client.put_object(
    Bucket=bucket_name,
    Key='raw/2026/03/marketing_campaigns.csv',
    Body=csv_buffer.getvalue()
)

# Also upload partitioned by channel (for faster queries)
for channel in df['channel'].unique():
    channel_df = df[df['channel'] == channel]
    channel_buffer = StringIO()
    channel_df.to_csv(channel_buffer, index=False)
    
    s3_client.put_object(
        Bucket=bucket_name,
        Key=f'processed/daily/channel={channel}/data.csv',
        Body=channel_buffer.getvalue()
    )
print(f"✅ Uploaded {len(df):,} records to S3")

# ============================================
# 4. AWS ATHENA SETUP - Serverless SQL Queries
# ============================================
print("\n🔍 Setting up AWS Athena for serverless queries...")

athena_client = boto3.client('athena', region_name='us-east-1')

# Create database in Athena
create_db_query = "CREATE DATABASE IF NOT EXISTS marketing_analytics;"

response = athena_client.start_query_execution(
    QueryString=create_db_query,
    QueryExecutionContext={'Database': 'marketing_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena-results/'}
)

print("✅ Athena database created")

# ============================================
# 5. CREATE EXTERNAL TABLE IN ATHENA
# ============================================
print("\n📊 Creating external table for marketing data...")

create_table_query = """
CREATE EXTERNAL TABLE IF NOT EXISTS marketing_campaigns (
    campaign_id INT,
    campaign_name STRING,
    channel STRING,
    campaign_type STRING,
    target_audience STRING,
    start_date DATE,
    end_date DATE,
    spend DOUBLE,
    impressions INT,
    clicks INT,
    conversions INT,
    revenue DOUBLE,
    roi DOUBLE,
    ctr DOUBLE,
    cpa DOUBLE
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe'
WITH SERDEPROPERTIES (
    'serialization.format' = ',',
    'field.delim' = ','
)
LOCATION 's3://{bucket_name}/raw/'
TBLPROPERTIES ('has_encrypted_data'='false');
"""

response = athena_client.start_query_execution(
    QueryString=create_table_query,
    QueryExecutionContext={'Database': 'marketing_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena-results/'}
)

# Wait for query to complete
execution_id = response['QueryExecutionId']
time.sleep(3)

print("✅ External table created - now you can query directly from S3!")

# ============================================
# 6. BUSINESS QUERIES IN ATHENA
# ============================================
print("\n💡 Running business queries in Athena...")

# Business Query 1: Channel performance summary
query1 = """
SELECT 
    channel,
    COUNT(*) as campaign_count,
    SUM(spend) as total_spend,
    SUM(revenue) as total_revenue,
    ROUND(AVG(roi), 2) as avg_roi
FROM marketing_campaigns
GROUP BY channel
ORDER BY avg_roi DESC;
"""

response = athena_client.start_query_execution(
    QueryString=query1,
    QueryExecutionContext={'Database': 'marketing_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena-results/'}
)
print("✅ Query 1 submitted: Channel performance")

# Business Query 2: Identify underperforming campaigns
query2 = """
SELECT 
    campaign_name,
    channel,
    spend,
    revenue,
    roi
FROM marketing_campaigns
WHERE roi < 0
ORDER BY roi
LIMIT 10;
"""

response = athena_client.start_query_execution(
    QueryString=query2,
    QueryExecutionContext={'Database': 'marketing_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena-results/'}
)
print("✅ Query 2 submitted: Negative ROI campaigns")

# Business Query 3: Monthly trends
query3 = """
SELECT 
    EXTRACT(year FROM start_date) as year,
    EXTRACT(month FROM start_date) as month,
    SUM(spend) as monthly_spend,
    SUM(revenue) as monthly_revenue,
    AVG(roi) as avg_roi
FROM marketing_campaigns
GROUP BY EXTRACT(year FROM start_date), EXTRACT(month FROM start_date)
ORDER BY year, month;
"""

response = athena_client.start_query_execution(
    QueryString=query3,
    QueryExecutionContext={'Database': 'marketing_analytics'},
    ResultConfiguration={'OutputLocation': f's3://{bucket_name}/athena-results/'}
)
print("✅ Query 3 submitted: Monthly trends")

# 7. GET QUERY RESULTS
# ============================================
def get_query_results(query_execution_id):
    """Helper function to get Athena query results"""
    while True:
        response = athena_client.get_query_execution(
            QueryExecutionId=query_execution_id
        )
        state = response['QueryExecution']['Status']['State']
        if state in ['SUCCEEDED', 'FAILED', 'CANCELLED']:
            break
        time.sleep(1)
    
    if state == 'SUCCEEDED':
        results = athena_client.get_query_results(
            QueryExecutionId=query_execution_id
        )
        return results
    return None
```
